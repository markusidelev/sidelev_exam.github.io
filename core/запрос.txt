select
        articles.title as title,
            articles.annotation as text,
            authors.name as name,
            articles.updated_at as date,
            photos.link as photo
        from
        `articles`,
            `authors`,
            `photos`
        where
            author_id = authors.id
            and
        author_id = 5
        and photo_id = photos.id
        and
        year(updated_at) = 2011
        and
        month(updated_at) > 8
        LIMIT 10
        OFFSET 0
